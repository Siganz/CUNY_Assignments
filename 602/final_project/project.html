def check_endpoint(base_api_url, app_token):
    headers = {
        'X-App-Token': app_token,
        'Content-Type': 'application/json'
    }
    
    # Test with small limit in URL
    url = f"{base_api_url}?$limit=1"
    
    response = requests.post(url, headers=headers)
    
    if response.status_code == 200:
        print("Endpoint is working")
        return True
    else:
        print(f"Endpoint failed. Status: {response.status_code}")
        print(f"Message: {response.text}")
        return False


def fetch_data(base_api_url, app_token):
    all_data = []
    offset = 0
    limit = 200000
    headers = {
        'X-App-Token': app_token,
        'Content-Type': 'application/json'
    }
    
    while True:
        url = f"{base_api_url}?$limit={limit}&$offset={offset}"
        
        response = requests.post(url, headers=headers)
        response.raise_for_status()
        
        batch = pd.DataFrame(response.json())
        all_data.append(batch)
        
        if len(batch) < limit:
            break
            
        offset += limit
    
    return pd.concat(all_data, ignore_index=True)


def load_or_fetch_data(base_api_url=None, csv_filename="nyc_payroll_data.csv.gz"):
    csv_path = Path(csv_filename)
    
    if csv_path.exists():
        return pd.read_csv(csv_path, low_memory=False)
    
    app_token = os.getenv("app_token")
    if not app_token:
        raise ValueError("app_token is missing from .env")
    
    if base_api_url is None:
        raise ValueError("base_api_url is required when fetching from API")
    
    df = fetch_data(base_api_url, app_token)
    df.to_csv(csv_path, index=False, compression="gzip")
    return df

def clean_money_column(series):
    """Clean a money column (remove $, commas, convert to float)"""
    return (series
            .astype(str)
            .str.replace('$', '', regex=False)
            .str.replace(',', '', regex=False)
            .str.strip()
            .replace('', pd.NA)
            .astype(float))
